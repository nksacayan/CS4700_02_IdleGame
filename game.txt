	Buttons
		// Training options
		*TEMPLATE
		on click:
			if (no inCombat)
				yield 1 xp
			else
				log Can't increase stats while in combat!
			end
		end

		  *trainButton
		    name:Train
		    on click:if (no inCombat) yield 1 strength

			*drinkButton
				name:Drink
				on click:if (no inCombat) yield 1 constitution

			*studyButton
		    name:Study
		    on click:if (no inCombat) yield 1 intelligence

			*meditateButton
				name:Meditate
				on click:if (no inCombat) yield 1 focus

		// Exploring
		*TEMPLATE
		//on click:if (have inCombat) log You're already in combat!

			*exploreForestButton
				name:Explore Forest
				on click:
					if (inCombat=0)
						inCombat=1
						enemyHealth=10
						enemyAttack=1
						log Found an enemy!
						show rabbitIcon
					end
				end

		*TEMPLATE

			// Combat
			*attackButton
				name:Attack
				tag:combatTag
				on click:
					if (inCombat)
						yield -(strength) enemyHealth
						yield -(enemyAttack) health
						if (no enemyHealth)
							log You have defeated the monster!
							hide rabbitIcon
							yield 5 gold
							yield 5 xp
							yield 5 renown
							inCombat=0
						end
					else
						log You are not in combat!
					end
				end

		// Icons
		*rabbitIcon
			tag:enemyIcon
			name:Evil Rabbit
			icon:https://imgur.com/7U428yg.jpg
