	Buttons
		// Training options
		*TEMPLATE
		on click:
			if (no inCombat)
				yield 1 xp
			else
				log Can't increase stats while in combat!
			end
		end

		  *trainButton
		    name:Train
		    on click:if (no inCombat) yield 1 strength

			*studyButton
		    name:Study
		    on click:if (no inCombat) yield 1 intelligence

			*drinkButton
				name:Drink
				on click:if (no inCombat) yield 1 constitution

			*meditateButton
				name:Meditate
				on click:if (no inCombat) yield 1 attunement

		// Exploring
		*TEMPLATE
		on click:if (have inCombat) log You're already in combat!

			*exploreForestButton
				name:Explore Forest
				on click:
					if (no inCombat)
						enemyHealth=10
						enemyAttack=1
						log Found an enemy!
						show rabbitIcon
						inCombat=1
					end
				end

		*TEMPLATE

			// Combat
			*attackButton
				name:Attack
				tag:combatTag
				on click:
					if (have inCombat)
						yield -(strength) enemyHealth
							if (no enemyHealth)
								log You have defeated the monster!
								inCombat=0
								hide rabbitIcon
							end
						end
					end
				end

		// Icons
		*rabbitIcon
			tag:enemyIcon
			name:Evil Rabbit
			icon:https://imgur.com/7U428yg.jpg
			hidden
			// no text

		// Memes
		*fuelButton

			on click:yield 1 fuel

		// Other
		*clearLogButton
			name:Clear Log
			on click:clear log

		*debugButton
			name:Debug Button
			desc:Does whatever I need to debug. Remove before actual game.
			on click:log inCombat=[inCombat]
