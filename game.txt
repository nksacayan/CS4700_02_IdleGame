	Buttons
		// Training options
			*TEMPLATE
				on click:
					if (no inCombat)
						yield 1 xp
					else
						log Can't increase stats while in combat!
					end
				end

			  *trainButton
			    name:Train
					desc:Increases Strength
			    on click:if (no inCombat) yield 1 str

				*drinkButton
					name:Drink
					desc:Increases Constitution
					on click:if (no inCombat) yield 1 con

				*studyButton
			    name:Study
					desc:Increases Intelligence
			    on click:if (no inCombat) yield 1 int

				*meditateButton
					name:Meditate
					desc:Increases Focus
					on click:if (no inCombat) yield 1 focus

		// Exploring
			*TEMPLATE
				on click:if (have inCombat) log You're already in combat!

				*exploreForestButton
					name:Explore Forest
					desc:Find monsters in the forest
					on click:
						if (inCombat=0)
							inCombat=1
							enemyHealth=10
							enemyAttack=1
							log Found an enemy!
							show rabbitIcon
						end
					end

		// Combat
			*TEMPLATE

				*attackButton
					name:Attack
					tag:combatTag
					on click:
						if (inCombat)
							yield -(str) enemyHealth
							yield -(enemyAttack) health
							if (no enemyHealth)
								log You have defeated the monster!
								hide rabbitIcon
								yield 5 gold
								yield 5 xp
								yield 5 renown
								inCombat=0
							end
						else
							log You are not in combat!
						end
					end

		// Icons
			*rabbitIcon
				tag:enemyIcon
				name:Evil Rabbit
				icon:https://imgur.com/7U428yg.jpg
